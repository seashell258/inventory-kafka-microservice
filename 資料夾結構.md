Get-ChildItem -Recurse -Force -ErrorAction SilentlyContinue |
    Where-Object { 
        $pathParts = $_.FullName.Split('\')
        -not ($pathParts -contains 'node_modules' -or 
              $pathParts -contains '.git' -or 
              $pathParts -contains '.expo' -or 
              $pathParts -contains 'dist')
    } |
    Sort-Object FullName |
    ForEach-Object {
        $level = $_.FullName.Split('\').Count - (Get-Location).Path.Split('\').Count
        ('  ' * $level) + '- ' + $_.Name
    }
